package "core" {

    class GameLoop {
        +run(): void
    }

    class GameSession {
        +sim: SimState
        +rng: RNG
        +bus: EventBus
        +turns: TurnManager
        +runStep(): void
    }

    class TurnManager {
        +currentSide: Side
        +actionCount: Int
        +startRound(): void
        +beginTurn(entity: Entity): void
        +commitAction(action: Action): Result
        +endTurn(): void
    }

    enum Side {
        Hunted
        Minotaur
    }

    class EventBus {
        +publish(evt: Event): void
        +subscribe(type: EventType, handler): void
    }

    enum EventType {
        TurnStarted
        ActionCommitted
        LootResolved
        MazeShifted
        EffectApplied
        InventoryChanged
        DespawnEntity
        CatchHunted
        AttackEntity
        GameStarted
        GameWon
        GameEnded
    }

    class Event {
        +type: EventType
        +payload: Map
        +timestamp: Long
    }

    class RNG {
        +seed(s: Long): void
        +uniformInt(lo: Int, hi: Int): Int
        +uniform01(): Float
        +rollD8(): Int
        +rollD8(quantity: Int): Int
    }
}

package "world" {

    enum Direction {
        North
        East
        South
        West
    }

    enum MazeShape {
        Rectangle
        Ellipse
    }

    enum MazeAlgorithm {
        None
        TestEasy
        TestMed
        TestHard
        DepthFirst
        Prim
        Kruskal
    }

    enum MazeStepType {
        Start
        Visit
        Consider
        Accept
        Reject
        Backtrack
        Complete
    }

    class Neighbor {
        +x: Int
        +y: Int
        +direction: Direction
        +valid: Boolean
    }

    class Cell {
        +x: Int
        +y: Int
        +active: Boolean
        +north: Boolean
        +east: Boolean
        +south: Boolean
        +west: Boolean
        +nbr: Neighbor[4]
        +nbrCount: Int
    }

    class MazeBuildConfig {
        +backingWidth: Int
        +backingHeight: Int
        +mazeWidth: Int
        +mazeHeight: Int
        +shape: MazeShape
        +algorithm: MazeAlgorithm
    }

    class MazeStep {
        +stepNumber: Int
        +type: MazeStepType
        +x: Int
        +y: Int
        +x2: Int
        +y2: Int
        +candidates: String
        +decision: String
        +mutation: String
        +result: String
    }

    class MazeStepPrinter {
        +toString(type: MazeStepType): String
        +printStep(step: MazeStep): void
    }

    class MazeState {
        +width(): Int
        +height(): Int
        +inBounds(x: Int, y: Int): Boolean
        +hasCell(x: Int, y: Int): Boolean
        +cellAt(x: Int, y: Int): Cell
        +finalizeNeighbors(): void
        +openWall(x: Int, y: Int, dir: Direction): void
        +closeWall(x: Int, y: Int, dir: Direction): void
        +toggleWall(x: Int, y: Int, dir: Direction): void
        +opposite(d: Direction): Direction
    }

    class MazeGenerator {
        +generate(config: MazeBuildConfig, steps: List<MazeStep>): MazeState
        +validateConfig(config: MazeBuildConfig): void
        +formShape(maze: MazeState, config: MazeBuildConfig): void
        +initializeClosedWalls(maze: MazeState): void
    }

    class MazeCarver {
        +carve(maze: MazeState, algorithm: MazeAlgorithm, steps: List<MazeStep>): void
    }
}
